00259         static float_type _DistD2( const CFEntry& lhs, const CFEntry& rhs )
00260         {
00261                 float_type dot = 0.0;
00262                 for(std::size_t i = 0 ; i < dim ; i++)
00263                         dot += lhs.sum[i] * rhs.sum[i];
00264 
00265                 float_type dist = ( rhs.n*lhs.sum_sq + lhs.n*rhs.sum_sq - 2*dot ) / (lhs.n*rhs.n);
00266                 //assert(dist >= 0.0);
00267                 return std::max(dist, 0.0);

00272         {
00273                 std::size_t tmpn = lhs.n+rhs.n;

00642         float_type average_dist_closest_pair_leaf_entries()
00643         {
00644                 std::size_t total_n = 0;
00645                 float_type      total_d = 0.0;
00646                 float_type      dist;
00647 
00648                 // determine new threshold
00649                 CFNodeLeaf* leaf = (CFNodeLeaf*)leaf_dummy.get();
00650                 while( leaf != NULL )
00651                 {
00652                         if( leaf->size >= 2 )
00653                         {
00654                                 std::vector<float_type> min_dists( leaf->size, (std::numeric_limits<float_type>::max)() );
00655                                 for( std::size_t i = 0 ; i < leaf->size - 1 ; i++ )
00656                                 {
00657                                         for( std::size_t j = i+1 ; j < leaf->size ; j++ )
00658                                         {

00660                                                 dist = dist >= 0.0 ? sqrt(dist) : 0.0;
00661                                                 if( min_dists[i] > dist )       min_dists[i] = dist;
00662                                                 if( min_dists[j] > dist )       min_dists[j] = dist;
00663                                         }
00664                                 }
00665                                 for( std::size_t i = 0 ; i < leaf->size ; i++ )
00666                                         total_d += min_dists[i];
00667                                 total_n += leaf->size;
00668                         }
