00176                 {

00179                         // prepare summaries for each subcluster
00180                         // summaries = ( center, radius, norm )
00181                         subsum_vec_type subclusters;
00182                         subclusters.reserve(entries.size());
00183                         for( std::size_t i = 0 ; i < entries.size() ; i++ )
00184                         {
00185                                 const CFEntry& e = entries[i];
00186                                 ublas_vec_type center(dim);
00187                                 std::copy(e.sum, e.sum + dim, center.begin());
00188                                 center /= e.n;
00189                                 subclusters.push_back( subcluster_summary( center, _Radius(e), std::sqrt(inner_prod(center, center) )) ); 
00190                         }
00191 
00192                         std::sort( subclusters.begin(), subclusters.end(), subcluster_lessthan_norm() );
00193 
00194                         // in addition to an individual summary for each subcluster
00195                         // calculate pairwise euclidean distances of subclusters
00196                         std::size_t n = subclusters.size();
00197                         ublas_sym_matrix_type dist_mat(n,n);
00198                         for( std::size_t i = 0 ; i < n-1 ; i++ )
00199                         {
00200                                 for( std::size_t j = i+1 ; j < n ; j++ )
00201                                 {
00202                                         ublas_vec_type diff = subclusters[i].center - subclusters[j].center;
00203                                         dist_mat(i,j) = inner_prod(diff, diff);
00204                                 }
